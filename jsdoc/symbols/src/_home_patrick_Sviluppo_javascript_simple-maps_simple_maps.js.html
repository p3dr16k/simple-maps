 34 /** Function for map initialization			 
 35  * @param myLat starting latitude for map
 36  * @param myLon starting longitude for map
 37  * @param myZoom starting zoom for map
 38  * @param showControls boolean flag for enable/disable default map controls
 39  * @param type type for map (0 = hybrid, 1 = roadmap, 2 = satellite, 3 = terrain)
 40  */
 41 function initialize(myLat, myLon, myZoom, type, showControls, newMinZoom, newMaxZoom) 
 42 {
 43 		var mapOptions = {
 44 		  center: { lat: myLat, lng: myLon},
 45 		  zoom: myZoom,
 46 		  mapTypeId: mapsId[type],
 47 		  disableDefaultUI: !showControls,
 48 		  minZoom: newMinZoom,
 49 		  maxZoom: newMaxZoom,		  
 50 		};
 51 		map = new google.maps.Map(document.getElementById('map-container'),	mapOptions);
 52 					
 53 		map.setTilt(0);
 54 					
 55 }

 73 /** Setter for map center
 74  * @param lat latitude of new center
 75  * @param lon longitude of new center
 76  */
 77 function setCenter(lat, lon, z)
 78 {
 79 	map.setCenter(new google.maps.LatLng(parseFloat(lat), parseFloat(lon)));
 80 	if(z !== -1)
 81 	{
 82 		map.setZoom(z);
 83 	}
 84 }

114 /** Get size of map
115  * @return size of a map
116  */
117 function getSize()
118 {
119 	var div = map.getDiv();
120 	var x = div.offsetWidth;
121 	var y = div.offsetHeight;
122 	return new google.maps.Point(x, y).toString().replace("(", "").replace(")","").replace(" ","");
123 }

125 /** Get bounds of map
126  * @return bounds of a map as a string
127  */
128 function getLatLngBounds()
129 {				
130 	var printable = map.getBounds().toString();
131 	
132 	while(printable.indexOf("(") > -1 || printable.indexOf(")") > -1 || printable.indexOf(" ") > -1)
133 	{
134 		printable = printable.replace("(", "").replace(")","").replace(" ", "");
135 	}
136     return printable;
137 }

148 /**
149  * Returns the zoom level at which the given rectangular region fits in the map view. 
150  * The zoom level is computed for the currently selected map type. 		
151  * @param {google.maps.LatLngBounds} bounds 
152  * @return {Number} zoom level
153 **/
154 function getBoundsZoomLevel(bounds)
155 {
156 	  var MAX_ZOOM = map.mapTypes.get( map.getMapTypeId() ).maxZoom || 21 ;
157 	  var MIN_ZOOM = map.mapTypes.get( map.getMapTypeId() ).minZoom || 0 ;
158 
159 	  var ne= map.getProjection().fromLatLngToPoint( bounds.getNorthEast() );
160 	  var sw= map.getProjection().fromLatLngToPoint( bounds.getSouthWest() ); 
161 
162 	  var worldCoordWidth = Math.abs(ne.x-sw.x);
163 	  var worldCoordHeight = Math.abs(ne.y-sw.y);
164 
165 	  //Fit padding in pixels 
166 	  var FIT_PAD = 40;
167 
168 	  for( var zoom = MAX_ZOOM; zoom >= MIN_ZOOM; --zoom )
169 	  { 
170 		  if( worldCoordWidth*(1<<zoom)+2*FIT_PAD < $(map.getDiv()).width() && 
171 			  worldCoordHeight*(1<<zoom)+2*FIT_PAD < $(map.getDiv()).height() )
172 		  {
173 			  return zoom;
174 		  }
175 	  }
176 	  return 0;
177 }

179 /** Function for add marker on map
180  * @param myTitle a title for marker
181  * @param latitude for marker
182  * @param longitude for marker
183  * @param img path of the image to use
184  * @param callback the callback invoked when marker is clicked
185 */
186 function drawMarker(myTitle, lat, lon, img, callback)
187 {
188 	//marker creation
189 	var marker = new google.maps.Marker({
190 		  position: new google.maps.LatLng(parseFloat(lat), parseFloat(lon)),
191 		  map: map,
192 		  title: myTitle,
193 		  icon: img
194     });
195 				  
196     //Setting the callback
197 	google.maps.event.addListener(marker, 'click', callback);				 
198 				 				  
199 }

201 /** Function for add marker on map
202  * @param myTitle a title for marker
203  * @param loc a string that rapresents a location
204  * @param img path of the image to use
205  * @param callback the callback invoked when marker is clicked
206 */
207 function drawMarkerByLoc(myTitle, loc, img, callback)
208 {	
209 	if(geocoder === null || geocoder === undefined)
210 	{
211 		geocoder = new google.maps.Geocoder();
212 	}
213 	var marker;
214 	geocoder.geocode( { 'address': loc}, function(results, status) 
215 		{
216 			if (status == google.maps.GeocoderStatus.OK) 
217 			{				
218 				map.setCenter(results[0].geometry.location);
219 				marker = new google.maps.Marker({
220 					map: map,
221 					position: results[0].geometry.location,
222 					tiyle: myTitle,
223 					icon: img
224 				});
225 				//Setting the callback
226 				google.maps.event.addListener(marker, 'click', callback);
227 			}
228 			else 
229 			{
230 				alert("Geocode was not successful for the following reason: " + status);
231 			}
232 		});	   				  
233 }

243 /**Function for polygon dawning
244  * @param coordinates String that rapresents an array of coordinates
245  * @param myStrokeColor String that rapresents stroke color of polygon
246  * @param myFillColor String that rapresents fill color of polygon
247  * @param myStrokeOpacity stroke opacity of polygon
248  * @param myStrokeWeight stroke weight of polygon			
249  * @param myFillOpacity fill opacity of polygon			
250  */
251 function drawPolygonByArray(coordinates, myStrokeColor, myFillColor, myStrokeOpacity, myStrokeWeight, myFillOpacity)
252 {
253 	var points = [];
254 	
255 	var tmpPointsArray = coordinates.split(",");
256 			
257 	var i = 0;
258 	var j = 0;
259 	while(i < tmpPointsArray.length - 1)
260 	{
261 		points[j] = new google.maps.LatLng(tmpPointsArray[i], tmpPointsArray[i + 1]);
262 		i = i + 2;
263 		j++;
264 	}
265 				
266 	var polygonToDraw = new google.maps.Polygon({
267 		paths: points,
268 		strokeColor: myStrokeColor,
269 		strokeOpacity: myStrokeOpacity,
270 		strokeWeight: myStrokeWeight,
271 		fillColor: myFillColor,
272 		fillOpacity: myFillOpacity
273 	});
274 				
275 	polygonToDraw.setMap(map);
276 }
